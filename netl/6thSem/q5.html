Write HTML code to demonstrate three types of lists: ordered , unordered & definition list.


<!DOCTYPE html>
<html lang="en">
<head>
<title>Question 5</title>
</head>
<body>
<!-- Ordered List -->
<ol>
<li>Apple</li>
<li>Mango</li>
<li>Grapes</li>
</ol>
<!-- Unordered List -->
<ul>
<li>Apple</li>
<li>Mango</li>
<li>Grapes</li>
</ul>
<!-- Definition List -->
<dl>
<dt>Apple</dt>
<dd>A red fruit</dd>
<dt>Mango</dt>
<dd>King of fruits</dd>
<dt>Grapes</dt>
<dd>sour in taste</dd>
</dl>
</body>
</html>
